266   * Post-JSON-mapping operation that populates the {@code messageTags} field "by hand".
267   * <p>

269   * can handle arbitrary Map types.

272   * The {@code JsonMapper} that was used to map to this type.
273   * @since 1.6.11
274   */
275  @JsonMappingCompleted
276  protected void jsonMappingCompleted(JsonMapper jsonMapper) {
277    if (rawMessageTags == null) {
278      return;
279    }
280    try {
281      messageTags = jsonMapper.toJavaList(rawMessageTags, MessageTag.class);
282      return;
283    } catch (FacebookJsonMappingException je) {
284      // message tags not in Graph API 2.5 format, ignore this exception and try another way
285    }
286    try {
287      JsonObject rawMessageTagsObject = jsonMapper.toJavaObject(rawMessageTags, JsonObject.class);
288      for (String key : getNames(rawMessageTagsObject)) {
289        String tagArrayString = rawMessageTagsObject.get(key).toString();
290        messageTags.addAll(jsonMapper.toJavaList(tagArrayString, MessageTag.class));
291      }
292      return;
293    } catch (FacebookJsonMappingException je) {
294    }
